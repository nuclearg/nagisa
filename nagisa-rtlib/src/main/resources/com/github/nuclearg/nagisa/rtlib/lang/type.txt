FUNCTION ITOA(i AS INTEGER) AS STRING
	DIM str AS STRING
	
	WHILE i > 0
		LET n = i % 10		
		LET i = i / 10
		
		LET str = CHR(n + ASC("0")) + str	
	END WHILE
	
	RETURN str
END FUNCTION
